DROP TABLE IF EXISTS images;
DROP TABLE IF EXISTS actions;
DROP TABLE IF EXISTS sessions;
DROP TABLE IF EXISTS users;

CREATE TABLE users(
id INT AUTO_INCREMENT PRIMARY KEY,
un VARCHAR(1000),
pw VARCHAR(1000)
);

CREATE TABLE sessions(
id INT AUTO_INCREMENT PRIMARY KEY,
user_id int,
sessiontoken varchar(1000),
expired tinyint,
time DateTime
);

CREATE TABLE actions(
id INT AUTO_INCREMENT PRIMARY KEY,
session_id int,
url varchar(1000),
content text,
time DateTime
);

CREATE TABLE images(
id INT AUTO_INCREMENT PRIMARY KEY,
user_id int,
name varchar(1000),
mimi varchar(1000),
image blob
);
